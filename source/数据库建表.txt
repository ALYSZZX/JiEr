供货商
create table supplier(
id int not null auto_increment,
name varchar(30) not null,
contacts varchar(30),
phone varchar(15),
address varchar(50),
remark varchar(200),
pay double,
defaults int,
disable int default 0,
primary key(id)
)charset=utf8;	


商品类型
create table category(
id int not null auto_increment,
category_number int not null,
catagory_name varchar(20),
primary key(id)
)charset=utf8;

商品表
create table goods(
goods_id int not null auto_increment,
goods_number int not null,
goods_name varchar(30) not null,
goods_category int,
goods_unit int,	
goods_spec varchar(20),
goods_color int,
goods_life int,
goods_status int,
goods_discount double,
goods_store int default 0,
goods_purchase double,
goods_sell double,
goods_integral double,
goods_production varchar(30),
goods_remark varchar(50),
primary key(goods_id),
constraint FK_goods_category foreign key(goods_category) references category(id),
constraint FK_goods_unit foreign key(goods_unit) references unit(id),
constraint FK_goods_color foreign key(goods_color) references color(id)  
)charset=utf8;
